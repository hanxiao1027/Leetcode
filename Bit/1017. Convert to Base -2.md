class Solution {
    public String baseNeg2(int N) {
        if(N == 0) return "0";
        int[] ori = new int[32];
        int[] ans = new int[37];
        String res = "";
        for(int i = 0; i < 32; i++){
            int bit = (N >> i) & 1;
            ori[i] = bit;
            
        }
        for(int i = 0; i < 32; i++){
            if(i % 2 == 0){
                ans[i] += ori[i];
            }
            else{
                ans[i + 1] += ori[i];
                ans[i] += ori[i];
            }
           // System.out.println(i + " " + ans[i]);
        }
        for(int i = 0; i < 35; i++){
            while(ans[i] > 1){
                    ans[i] -= 2;
                    ans[i + 1]++;
                    ans[i + 2] ++;  
            }
        }
        boolean start = false;
        for(int i = 33; i >= 0; i--){
            if(!start){
                if(ans[i] == 1){
                    start = true;
                    res = res + '1';
                }
            }
            else{
                System.out.println(ans[i]);
                res = res + ans[i];
            }

        }
        return res;
        
    }
}
