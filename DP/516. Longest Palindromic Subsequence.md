class Solution {
    public int longestPalindromeSubseq(String s) {
        if(s == null || s.length() == 0) return 0;
        int len = s.length();
        if(len == 1) return 1;
        int[][] dp = new int[len][len];
        for(int i = 0; i < len; i++){
            dp[i][i] = 1;
        }
        for(int i = 0; i < len - 1; i ++){
            dp[i][i + 1] = (s.charAt(i) == s.charAt(i + 1))? 2 : 1;
            //System.out.println(i + " " + dp[i][i + 1]);
        }
        if(len < 3){
            return dp[0][1];
        }
        for(int k = 2; k < len; k++){
            for(int i = 0; i + k < len; i++){
                int j = i + k;
                int max = dp[i][j - 1];
                for(int curr = i; curr < j; curr++){
                    if(s.charAt(curr) == s.charAt(j)){
                        max = Math.max(dp[curr + 1][j - 1] + 2, max);
                        //System.out.println(curr+ " " + j + " " + max);
                        break;
                    }
                }
                dp[i][j] = max;
                //System.out.println(i + " " + j + " " + max);
                
            }
        }
        return dp[0][len - 1];

        
    }
}
